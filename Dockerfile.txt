FROM node:20-bullseye-slim

RUN apt-get update && \
    apt-get install -y git ffmpeg imagemagick webp curl && \
    rm -rf /var/lib/apt/lists/*

RUN git clone https://github.com/usamile269/AHMAD_MD_X_MINI2

WORKDIR /root/ahmed-bot

# ✅ Persistent session directory — mount this as Volume on Railway: /data
RUN mkdir -p /data/sessions

ENV SESSION_DIR=/data/sessions

RUN npm install && npm install -g pm2

EXPOSE 8080

CMD ["npm", "start"]
